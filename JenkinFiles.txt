pipeline {
    agent any
     stages {
        stage('Prepare') {
            steps {
                // Get some code from a GitHub repository
                 git 'https://github.com/GarimaGitLearn/jenkinsIntegration.git'
           }
        }
         stage('Build') {
           // Run the maven build
            steps {
                bat "mvn clean test"
            }
        
        }
         stage('Package') {
            // Run the maven build
            steps {
                bat "mvn package"
            }
        }
         stage('Results') {
            steps{
                post {
                        success {
                        junit '**/target/surefire-reports/TEST-*.xml'
                        archiveArtifacts 'target/*.jar'
                    
                    }
                }
            }
        }
    }
}
